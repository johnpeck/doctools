%------------------ Define the fancy page styles ----------------------

% first page style
%
% \firstlefthead, \firstrighthead, \firstrightfoot, \firstleftfoot
% commands must be defined in the document.
\fancypagestyle{first}{
    % The rules at the page top and bottom
    \renewcommand{\headrulewidth}{0.4pt}
    \renewcommand{\footrulewidth}{0.4pt}
    \lhead{\firstlefthead{}}
    \rhead{\firstrighthead{}}
    \rfoot{\firstrightfoot{}}
    \cfoot{}
    \lfoot{\firstleftfoot{}}
}




% normal page style
\fancypagestyle{normal}{
    % The rules at the page top and bottom
    \renewcommand{\headrulewidth}{0.4pt}
    \renewcommand{\footrulewidth}{0.4pt}
    \lhead{\nouppercase \leftmark}
    \rhead{\nouppercase \rightmark}
    \rfoot{\normalrightfoot{}}
    \cfoot{\thepage{}}
    \lfoot{\normalleftfoot{}}
} % End of normal page definition
