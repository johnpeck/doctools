% Generalized way to enter data into a table and format the cell.
% Formats the cell height in a nice way.
% Usage: \wksentry{<cell width>}{<Text in cell>}
\newcommand{\wksentry}[2]{
        \begin{minipage}[c][\totalheight+.5cm][c]{#1}
        \raggedright
        #2
        \end{minipage}
}%end wksentry


% cwksentry() 
% 
% Another nice way to put data in a cell.  Centers the text
% in a minipage.
%  
% Usage: \cwksentry{<cell width>}{<Text in cell>}
\newcommand{\cwksentry}[2]{
  \begin{minipage}[c][\totalheight+.5cm][c]{#1}
    \begin{center}
      #2
    \end{center}
  \end{minipage}
} % end cwksentry

% shortboxtext()
% 
% Text inside a box shortened for use in an itemized environment.
\newcommand{\shortboxtext}[1]{
  \vspace{-.5\baselineskip}
  \begin{flushright}
    \singlespacing
    \vspace{-\baselineskip}
    \fbox{
      \begin{minipage}[t]{\textwidth - 1cm - 2\tabcolsep }
        \noindent
        {#1}
      \end{minipage}
    }
  \end{flushright}
}% end \shortboxtext

% onecodeline()
%
% Uses \path from the url package to verbatim-print input all on one
% line.  This is handy if you have a long line of input for verbatim
% that you'd like to put on two lines in your editor, but not in the output.
\newcommand{\onecodeline}[1]{
  \vspace{-\baselineskip}
  \begin{center}
    \colorbox{codegray}{
      \begin{minipage}[c]{\textwidth - 2\tabcolsep}
        \path{#1}
      \end{minipage}
    }
  \end{center}
}


% filesnip()
%
% Show a snippet of a file's contents in a box with the filename as
% the title.  The filename is shaded.
%
% Requires: colortbl
%
% Usage: \filesnip{<filename>}{<snippet>}
\newcommand{\filesnip}[2]{
\begin{center}
    \vspace{-\baselineskip}
    \begin{tabular}{|l|} \hline
        \rowcolor[gray]{0.8}
        \begin{minipage}[c]{\textwidth - 2\tabcolsep}
            \textbf{File:}
            #1
        \end{minipage}\\
        \begin{minipage}[c]{\textwidth - 2\tabcolsep}
            \vspace{0.5\baselineskip}
            $\cdots$ \\
            \begin{minipage}[c]{\textwidth - 2\tabcolsep}
                #2
            \end{minipage}
            $\cdots$\\
            \vspace{-0.5\baselineskip}
        \end{minipage}\\
    \hline
    \end{tabular}
\end{center}
}

% Show the entire contents of a file (not a snippet) in a box with the
% filename as the title.  The filename is shaded.
% Usage: \filecont{<filename>}{<text>}
\newcommand{\filecont}[2]{
\begin{center}
    \vspace{-\baselineskip}
    \begin{tabular}{|l|} \hline
        \rowcolor[gray]{0.8}
        \begin{minipage}[c]{\textwidth - 2\tabcolsep}
            \textbf{File:}
            #1 (Entire file)
        \end{minipage}\\
        \begin{minipage}[c]{\textwidth - 2\tabcolsep}
        \singlespacing
        #2
        \vspace{1mm}
        \end{minipage}\\
    \hline
    \end{tabular}
\end{center}
}

% An ascii command definition.  Just places the command in typewriter
% font on one side of the page, and its description on the other.
% Usage: \asciicmd{<command>}{<description}
%
% Very important: if you want to add a label or index entry for the
% command, put the \label or \index command inside the {<description>}
% argument.  Otherwise the \label or \index commands may end up on the
% page preceeding the minipage -- the index refers to the ascii
% command on the page before it actually appears.
\newcommand{\asciicmd}[2]{
    \noindent
    \begin{tabular}{@{}l@{}c@{}}
    \begin{minipage}[c][\totalheight+1cm][c]{\cmdwidth}
        \texttt{#1}
    \end{minipage}
    &
    \begin{minipage}[c][\totalheight+1cm][c]{\textwidth-\cmdwidth}
    #2
    \end{minipage}
    \end{tabular}
}

% Make a figure sitting against the left margin that text wraps around.
%
% The number of shortened text lines produced must be specified to avoid
% conflicts with floats below the figure.  Change this number if your
% floats don't move over to the left margin.
%
% Usage: \sidewrap{<image>}{<scale factor>}{<number of short lines>}
% Must define \sidefigwidth somewhere.  A nice value is 6cm
\newcommand{\sidewrap}[3]{
    \begin{wrapfigure}[#3]{L}[0cm]{\sidefigwidth}
        \begin{minipage}[c][]{\sidefigwidth}
            \begin{center}
                \includegraphics[clip,scale=#2]{#1}
            \end{center}
        \end{minipage}
    \end{wrapfigure}
}

% Make two pieces of text -- one up against the left margin, and the

\newcommand{\leftright}[2]{

            {\noindent #2}
            \end{flushright}
        \end{minipage}\\
    \end{tabular}
}
